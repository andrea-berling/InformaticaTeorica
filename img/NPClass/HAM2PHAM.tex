\newcommand{\GraphG}[3]{% x,y,name
    \node[point] (#3_m1) at ([shift={(#1 cm,#2 cm)}]0,0) {};
    \node[point] (#3_m2) [below=2cm of #3_m1] {};
    \node[point] (#3_m3) [right=2cm of #3_m2] {};
    \node[point] (#3_m4) [above=2cm of #3_m3] {};
    \draw (#3_m3) -- (#3_m1) -- (#3_m2) -- (#3_m3) -- (#3_m4) -- (#3_m2);
}

\begin{document}
    \begin{tikzpicture}[point/.style={draw,circle,inner sep=0cm,fill=black,minimum size=2pt}]
        \GraphG{0}{0}{G}
        \draw[decorate,decoration=snake,->] (3,-1) -- (4,-1);
        \GraphG{5}{0}{G2}
        \node[point] (n1) at (6,1) {};
        \node[point] (start) [right=1cm of n1];
        \node[point] (n2) at (6,-3) {};
        \node[point] (end) [right=1cm of n2];
        \node () [left=2pt of n1] {$n_{1}$};
        \node () [right=2pt of start] {$n_{\textsc{start}}$};
        \node () [left=2pt of n2] {$n_{2}$};
        \node () [right=2pt of end] {$n_{\textsc{end}}$};
        \draw (start) -- (n1);
        \draw (end) -- (n2);
        \foreach \n in {1,...,4}
        {
            \draw (n1) -- (G2_m\n);
            \draw (n2) -- (G2_m\n);
        }
    \end{tikzpicture}
\end{document}
