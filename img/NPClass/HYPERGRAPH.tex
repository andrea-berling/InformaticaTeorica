\newcommand{\GraphG}[3]{% x,y,name
    \node[point] (#3_m1) at ([shift={(#1 cm,#2 cm)}]0,0) {};
    \node[point] (#3_m2) [below=2cm of #3_m1] {};
    \node[point] (#3_m3) [right=2cm of #3_m2] {};
    \node[point] (#3_m4) [above=2cm of #3_m3] {};
    \draw (#3_m3) -- (#3_m1) -- (#3_m2) -- (#3_m3) -- (#3_m4) -- (#3_m2);
}
